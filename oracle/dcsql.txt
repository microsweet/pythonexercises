SELECT c.XZQMC, t.OPERATION_OBJ OPERATION_OBJ, n.LANDUSER, p.NAME_, n.RISKRESOURCE, n.RISKLEVEL, n.RISKDEPICT
FROM GRID_BUSINESS.BS_TASK_P_TASKS t, GRID_SYSDB.COUNTY c, GRID_SYSDB.BBS_RISKLAND n, (SELECT DISTINCT name_, PROC_INST_ID_ FROM GRID_ACTIVIT.ACT_RU_TASK) p
WHERE t.TASK_TYPE=2 AND SUBSTR(t.FJBM, 2,6)=c.XZBM AND t.OPERATION_OBJ=n.CADASTRALNO AND t.PROC_INST_ID=p.PROC_INST_ID_(+)
AND NOT EXISTS (SELECT * FROM GRID_SYSDB.BBS_DRAWRISK m WHERE t.OPERATION_OBJ=m.CADASTRALNO)
ORDER BY c.XZQMC, t.OPERATION_OBJ
